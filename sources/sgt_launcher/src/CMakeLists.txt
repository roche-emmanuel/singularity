SET(TARGET_NAME "sgtLauncher")

FILE(GLOB_RECURSE PUBLIC_HEADERS "../include/*.h")
FILE(GLOB_RECURSE SOURCE_FILES "launcher.cpp")

INCLUDE_DIRECTORIES(../../sgtCore/include)

ADD_RESSOURCE(launcher.rc SOURCE_FILES)

IF(MSVC)
ADD_DEFINITIONS(-DSGT_WINDOWS)
ADD_EXECUTABLE (${TARGET_NAME} WIN32 ${SOURCE_FILES})
ELSE()

ENDIF()

TARGET_LINK_LIBRARIES(${TARGET_NAME} sgtCore ${LUA_LIBS} ${OSG_LIBS} ${FLAVOR_LIBS})

INSTALL(TARGETS ${TARGET_NAME} DESTINATION bin/${BIN_FOLDER}/)

COMPRESS_BINARY()
