SET(TARGET_NAME "sgtProland")

INCLUDE_DIRECTORIES (../include)
INCLUDE_DIRECTORIES (../../sgtCore/include)
INCLUDE_DIRECTORIES (../include/core)
INCLUDE_DIRECTORIES (../include/terrain)

INCLUDE_DIRECTORIES(${GLEW_INC_DIR})
LINK_DIRECTORIES (${GLEW_LIB_DIR})

INCLUDE_DIRECTORIES(${PTHREADS_INC_DIR})
LINK_DIRECTORIES (${PTHREADS_LIB_DIR})

INCLUDE_DIRECTORIES(${GLUT_INC_DIR})
LINK_DIRECTORIES (${GLUT_LIB_DIR})

INCLUDE_DIRECTORIES(${ATB_INC_DIR})
LINK_DIRECTORIES (${ATB_LIB_DIR})

INCLUDE_DIRECTORIES(${TIFF_INC_DIR})
LINK_DIRECTORIES (${TIFF_LIB_DIR})

ADD_DEFINITIONS(-DSGTPROLAND_LIB)
ADD_DEFINITIONS(-DTIXML_USE_STL)
ADD_DEFINITIONS(-D_CRT_SECURE_NO_WARNINGS)
ADD_DEFINITIONS(${OSG_FLAGS})
ADD_DEFINITIONS(${GLEW_FLAGS})

IF(MSVC)
	SET(CMAKE_SHARED_LINKER_FLAGS "${CMAKE_SHARED_LINKER_FLAGS} /NODEFAULTLIB:LIBCMT")
ENDIF()

FILE(GLOB_RECURSE PUBLIC_HEADERS "../include/*.h")

FILE(GLOB_RECURSE SOURCE_FILES "*.cpp" )

ADD_MSVC_PRECOMPILED_HEADER("sgtProland.h" "sgtProland.cpp" SOURCE_FILES)

ADD_LIBRARY (${TARGET_NAME} SHARED ${PUBLIC_HEADERS} ${SOURCE_FILES})

TARGET_LINK_LIBRARIES(${TARGET_NAME} sgtCore ${LUA_LIBS} ${GL_LIBS} ${GLUT_LIBS} ${BOOST_LIBS} ${PTHREADS_LIBS} ${ATB_LIBS} ${GLEW_LIBS} ${OSG_LIBS} ${TIFF_LIBS})

INSTALL_PDB()

INSTALL(TARGETS ${TARGET_NAME}
	RUNTIME DESTINATION bin/${FLAVOR}
	LIBRARY DESTINATION bin/${FLAVOR}
	ARCHIVE DESTINATION ../sources/${TARGET_NAME}/lib)

COMPRESS_BINARY()
