SET(TARGET_NAME "sgtLand")

INCLUDE_DIRECTORIES (../include)
INCLUDE_DIRECTORIES (../../sgtCore/include)

ADD_DEFINITIONS(-DSGTLAND_LIB)
ADD_DEFINITIONS(-D_CRT_SECURE_NO_WARNINGS)
ADD_DEFINITIONS(${OSG_FLAGS})

FILE(GLOB_RECURSE PUBLIC_HEADERS "../include/*.h")

FILE(GLOB_RECURSE SOURCE_FILES "*.cpp" )


ADD_MSVC_PRECOMPILED_HEADER("sgtLand.h" "sgtLand.cpp" SOURCE_FILES)

ADD_LIBRARY (${TARGET_NAME} SHARED ${PUBLIC_HEADERS} ${SOURCE_FILES})

TARGET_LINK_LIBRARIES(${TARGET_NAME} sgtCore ${LUA_LIBS} ${GL_LIBS} ${BOOST_LIBS} ${OSG_LIBS})

INSTALL_PDB()

INSTALL(TARGETS ${TARGET_NAME}
	RUNTIME DESTINATION bin/${FLAVOR}
	LIBRARY DESTINATION bin/${FLAVOR}
	ARCHIVE DESTINATION ../sources/${TARGET_NAME}/lib)

COMPRESS_BINARY()
